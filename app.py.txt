import streamlit as st
import requests
import pandas as pd
import plotly.graph_objects as go

st.set_page_config(page_title="Deribit GEX Monitor", layout="wide")

# --- BUSCA DE DADOS ---
@st.cache_data(ttl=60) # Atualiza a cada 60 segundos
def get_deribit_data(coin):
    # Pega o preço do índice
    idx_url = f"https://www.deribit.com/api/v2/public/get_index_price?index_name={coin.lower()}_usd"
    price = requests.get(idx_url).json()['result']['index_price']
    
    # Pega o resumo de todos os contratos de opções
    summary_url = f"https://www.deribit.com/api/v2/public/get_book_summary_by_currency?currency={coin}&kind=option"
    data = requests.get(summary_url).json()['result']
    
    df = pd.DataFrame(data)
    # O nome do instrumento na Deribit é: BTC-28MAR26-50000-C
    # Vamos extrair o Strike e o Tipo (C/P) do nome
    df['strike'] = df['instrument_name'].str.split('-').str[2].astype(float)
    df['type'] = df['instrument_name'].str.split('-').str[3]
    
    return df, price

# --- INTERFACE ---
st.title("⚡ Deribit Real-Time GEX")
moeda = st.sidebar.selectbox("Moeda", ["BTC", "ETH", "SOL"])

try:
    df_opcoes, preco_spot = get_deribit_data(moeda)
    
    # Filtro de Strikes ao redor do preço
    margem = preco_spot * 0.15 # 15% de variação
    df_filt = df_opcoes[(df_opcoes['strike'] > preco_spot - margem) & (df_opcoes['strike'] < preco_spot + margem)].copy()

    # Cálculo do GEX (A Deribit já fornece o 'gamma' na API!)
    # GEX = Gamma * Open Interest * Preço do Spot (para escala financeira)
    df_filt['gamma'] = df_filt['gamma'].fillna(0)
    df_filt['gex_total'] = df_filt.apply(lambda r: r['gamma'] * r['open_interest'] * preco_spot if r['type'] == 'C' 
                                        else -r['gamma'] * r['open_interest'] * preco_spot, axis=1)

    # Agrupar por Strike
    final_gex = df_filt.groupby('strike')['gex_total'].sum().reset_index()
    final_gex['color'] = ['#00FF00' if x > 0 else '#FF0000' for x in final_gex['gex_total']]

    # --- GRÁFICO ---
    fig = go.Figure()
    fig.add_trace(go.Bar(x=final_gex['gex_total'], y=final_gex['strike'], orientation='h', marker_color=final_gex['color']))
    fig.add_hline(y=preco_spot, line_dash="dash", line_color="yellow", annotation_text=f"SPOT: {preco_spot:.0f}")
    
    fig.update_layout(template="plotly_dark", height=700, title=f"Gamma Exposure Financeiro - {moeda}")
    st.plotly_chart(fig, use_container_width=True)

except Exception as e:
    st.error(f"Erro ao conectar com a Deribit: {e}")

import streamlit as st
import requests
import pandas as pd
import plotly.graph_objects as go

# Função para buscar dados reais da Deribit
def buscar_dados(moeda):
    # Pega o preço atual do Bitcoin/Ethereum
    url_preco = f"https://www.deribit.com/api/v2/public/get_index_price?index_name={moeda.lower()}_usd"
    preco = requests.get(url_preco).json()['result']['index_price']
    
    # Pega o resumo de todas as opções (Gamma, Open Interest, etc)
    url_summary = f"https://www.deribit.com/api/v2/public/get_book_summary_by_currency?currency={moeda}&kind=option"
    dados = requests.get(url_summary).json()['result']
    
    df = pd.DataFrame(dados)
    # Extrai o Strike do nome do contrato (Ex: BTC-26DEC25-50000-C -> 50000)
    df['strike'] = df['instrument_name'].str.split('-').str[2].astype(float)
    df['tipo'] = df['instrument_name'].str.split('-').str[3] # C para Call, P para Put
    
    return df, preco

st.set_page_config(page_title="Deribit GEX", layout="wide")
st.title("📊 Monitor de Gamma - Deribit")

moeda_selecionada = st.sidebar.selectbox("Escolha a Moeda", ["BTC", "ETH", "SOL"])

if st.sidebar.button("Carregar Dados"):
    df_opcoes, preco_spot = buscar_dados(moeda_selecionada)
    
    # Filtramos apenas strikes próximos ao preço atual (para o gráfico não ficar gigante)
    margem = preco_spot * 0.20 # Mostra 20% acima e abaixo
    df_filt = df_opcoes[(df_opcoes['strike'] > preco_spot - margem) & (df_opcoes['strike'] < preco_spot + margem)].copy()

    # Cálculo do GEX: Gamma * Open Interest * Preço Spot
    # Se for Put, multiplicamos por -1 para mostrar a pressão negativa
    df_filt['gex_calculado'] = df_filt.apply(
        lambda r: (r['gamma'] * r['open_interest'] * preco_spot) if r['tipo'] == 'C' 
        else -(r['gamma'] * r['open_interest'] * preco_spot), axis=1
    )

    # Somamos tudo por Strike
    gex_final = df_filt.groupby('strike')['gex_calculado'].sum().reset_index()
    gex_final['cor'] = ['#00FF00' if x > 0 else '#FF0000' for x in gex_final['gex_calculado']]

fig = go.Figure()
    fig.add_trace(go.Bar(
        x=gex_final['gex_calculado'],
        y=gex_final['strike'],
        orientation='h',
        marker_color=gex_final['cor']
    ))
    
    # Linha amarela indicando o preço atual do mercado
    fig.add_hline(y=preco_spot, line_dash="dash", line_color="yellow", annotation_text="PREÇO ATUAL")

    fig.update_layout(template="plotly_dark", height=800, title=f"Exposição Gamma Líquida - {moeda_selecionada}")
    st.plotly_chart(fig, use_container_width=True)
